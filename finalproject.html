Final Project: 桌面智能光线检测仪

ESP32 + 光敏电阻采集桌面光照，以红绿 LED、TFT 显示屏和网页服务形成三重反馈。

第十周：期末创新项目项目档案项目介绍：本项目是一款基于ESP32的桌面智能光线检测仪，通过光敏电阻实时采集环境光照强度，利用红绿LED灯与显示屏三重反馈机制，直观提示用户当前光线是否过暗、适宜或过亮，旨在为宿舍学生提供低成本、无感的护眼辅助工具。核心用户：经常在书桌前学习或办公的学生、办公人员使用场景：室内办公环境、宿舍书桌前、图书馆书桌前核心方案输入：光敏电阻采集的环境光线强度模拟信号 / 环境光强数据处理逻辑：光线传感器→ADC转换→ESP32主控程序判断：适宜范围→绿灯；过暗→红灯；过亮→OLED显示橙色提示输出反馈：绿灯（光线适宜）、红灯（光线过暗）、显示屏橙色提示（光线过亮）硬件清单 & 结构材料类型器件名称备注主控ESP32 开发板Wi-Fi + 低功耗控制传感器光敏电阻模拟量/数字量输出，实时监测环境光强显示小尺寸OLED显示屏显示光线过亮/正常/过暗提示交互/反馈红绿双色 LED灯直观反馈光线状态（过暗/适宜/过亮）交互/反馈蜂鸣器（备选）声音提醒（备选）电源5V/2A 充电宝或电池盒保证 ESP32 与显示屏稳定供电结构3D打印外壳PLA材质，预留传感器与屏幕窗口辅材杜邦线、面包板防漏水、稳固连接项目设计日志周五上午：头脑风暴&项目设计卡团队经过头脑风暴，最终达成一致，完成《Hackathon 项目设计卡》。最初设想该装置具备声音提醒功能，但由于蜂鸣器快递运输时效延误，故将蜂鸣器提醒功能删除。

完整代码（LED灯 + 显示屏）
#include <WiFi.h>
#include <WebServer.h>
#include <Adafruit_GFX.h>
#include <Adafruit_ST7735.h>
#include <SPI.h>

// ========== WiFi配置 ==========
const char* ssid     = "你的WiFi名称";
const char* password = "你的WiFi密码";

// ========== 硬件引脚定义 ==========
// TFT屏幕引脚
#define TFT_CS     15
#define TFT_RST    4
#define TFT_DC     5
#define TFT_SCK    18
#define TFT_MOSI   23

// LED引脚
const int ledGreen = 13;
const int ledRed   = 12;

// 光敏引脚
const int lightPin = 34;

// 光照阈值
const int darkThreshold  = 1500;
const int brightThreshold= 3000;

// 屏幕与网页全局变量
Adafruit_ST7735 tft = Adafruit_ST7735(TFT_CS, TFT_DC, TFT_RST);
WebServer server(80);
int lightValue = 0;
String statusText = "";
uint16_t statusColor;

// 颜色定义
#define RED     ST7735_RED
#define GREEN   ST7735_GREEN
#define ORANGE  0xFC60

// ========== 网页服务函数 ==========
void handleRoot() {
  String html = "<!DOCTYPE html><html lang='zh-CN'>";
  html += "<head><meta charset='UTF-8'><title>桌面光照监测</title>";
  html += "<meta http-equiv='refresh' content='2'>";
  html += "<style>body{text-align:center;font-family:微软雅黑;margin-top:50px;}.box{width:420px;margin:0 auto;padding:30px;border:1px solid #eee;border-radius:12px;}.val{font-size:32px;font-weight:bold;}.tip{font-size:20px;margin-top:20px;}</style>";
  html += "</head><body><div class='box'>";
  html += "<h2>ESP32-S2 桌面光照监测系统</h2>";
  html += "<p>当前光照采样值：<span class='val'>" + String(lightValue) + "</span></p>";
  html += "<p class='tip'>当前状态：" + statusText + "</p>";
  html += "</div></body></html>";
  server.send(200, "text/html", html);
}

void setup() {
  Serial.begin(115200);

  // LED引脚初始化
  pinMode(ledGreen, OUTPUT);
  pinMode(ledRed, OUTPUT);
  digitalWrite(ledGreen, LOW);
  digitalWrite(ledRed, LOW);

  // TFT屏幕初始化
  tft.initR(INITR_BLACKTAB);
  tft.fillScreen(ST7735_BLACK);
  tft.setTextSize(1);
  tft.setTextColor(ST7735_WHITE);
  tft.setCursor(10, 20);
  tft.println("光照监测系统启动...");
  tft.display();
  delay(1000);

  // 连接WiFi
  Serial.print("正在连接WiFi: ");
  Serial.println(ssid);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi连接成功！");
  Serial.print("设备访问IP：");
  Serial.println(WiFi.localIP());

  server.on("/", handleRoot);
  server.begin();
}

void loop() {
  server.handleClient();

  lightValue = analogRead(lightPin);
  Serial.print("光照数值：");
  Serial.println(lightValue);

  digitalWrite(ledRed, LOW);
